371  /**
372   * @see com.restfb.FacebookClient#deleteObject(java.lang.String, com.restfb.Parameter[])
373   */
374  @Override
375  public boolean deleteObject(String object, Parameter... parameters) {
376    verifyParameterPresence("object", object);
377
378    String responseString = makeRequest(object, true, true, null, parameters);
379    try {
380      boolean isSuccess = false;
381      JsonObject jObj = new JsonObject(responseString);
382      if (jObj.has("success")) {
383        isSuccess = jObj.getBoolean("success");
384      }
385      if (jObj.has("result")) {
386        isSuccess = jObj.getString("result").contains("Successfully deleted");
387      }
388      return isSuccess;
389    } catch (JsonException jex) {
390      if (CLIENT_LOGGER.isTraceEnabled()) {
391        CLIENT_LOGGER.trace("no valid JSON returned while deleting a object, using returned String instead", jex);
392      }
393    }
394
395    return "true".equals(responseString);
396  }

413    String connectionJson;
414    if (!isBlank(accessToken) && !isBlank(appSecret)) {
415      connectionJson = makeRequestAndProcessResponse(new Requestor() {
416        @Override
417        public Response makeRequest() throws IOException {
418          return webRequestor.executeGet(String.format("%s&%s=%s", connectionPageUrl,
419            urlEncode(APP_SECRET_PROOF_PARAM_NAME), obtainAppSecretProof(accessToken, appSecret)));
420        }
421      });
422    } else {

424        @Override
425        public Response makeRequest() throws IOException {
426          return webRequestor.executeGet(connectionPageUrl);
427        }
428      });
429    }
430
431    return new Connection<T>(this, connectionJson, connectionType);

444  /**
445   * @see com.restfb.FacebookClient#fetchObjects(java.util.List, java.lang.Class, com.restfb.Parameter[])
446   */
447  @Override
448  @SuppressWarnings("unchecked")
449  public <T> T fetchObjects(List<String> ids, Class<T> objectType, Parameter... parameters) {
450    verifyParameterPresence("ids", ids);
451    verifyParameterPresence("connectionType", objectType);
452
453    if (ids.isEmpty()) {
454      throw new IllegalArgumentException("The list of IDs cannot be empty.");
455    }
456
457    for (Parameter parameter : parameters) {
458      if (IDS_PARAM_NAME.equals(parameter.name)) {
459        throw new IllegalArgumentException("You cannot specify the '" + IDS_PARAM_NAME + "' URL parameter yourself - "
460            + "RestFB will populate this for you with " + "the list of IDs you passed to this method.");
461      }
462    }
463
464    // Normalize the IDs
465    for (int i = 0; i < ids.size(); i++) {
466      String id = ids.get(i).trim();
467      if ("".equals(id)) {
468        throw new IllegalArgumentException("The list of IDs cannot contain blank strings.");
469      }
470
471      ids.set(i, id);
472    }
473
474    try {
475      String jsonString =
476          makeRequest("", parametersWithAdditionalParameter(Parameter.with(IDS_PARAM_NAME, join(ids)), parameters));
477
478      return jsonMapper.toJavaObject(jsonString, objectType);
479    } catch (JsonException e) {
480      throw new FacebookJsonMappingException("Unable to map connection JSON to Java objects", e);
481    }
482  }

533    }
534
535    try {

538
539      JsonObject normalizedJson = new JsonObject();
540
541      for (int i = 0; i < jsonArray.length(); i++) {
542        JsonObject jsonObject = jsonArray.getJsonObject(i);
543
544        // For empty resultsets, Facebook will return an empty object instead of
545        // an empty list. Hack around that here.
546        JsonArray resultsArray = jsonObject.get("fql_result_set") instanceof JsonArray
547            ? jsonObject.getJsonArray("fql_result_set") : new JsonArray();
548
549        normalizedJson.put(jsonObject.getString("name"), resultsArray);
550      }
551
552      return objectType.equals(JsonObject.class) ? (T) normalizedJson
553          : jsonMapper.toJavaObject(normalizedJson.toString(), objectType);
554    } catch (JsonException e) {
555      throw new FacebookJsonMappingException("Unable to process fql.multiquery JSON response", e);
556    }

865  @Override
866  @SuppressWarnings("unchecked")
867  public <T> T parseSignedRequest(String signedRequest, String appSecret, Class<T> objectType) {
868    verifyParameterPresence("signedRequest", signedRequest);
869    verifyParameterPresence("appSecret", appSecret);
870    verifyParameterPresence("objectType", objectType);
871
872    String[] signedRequestTokens = signedRequest.split("[.]");
873
874    if (signedRequestTokens.length != 2) {
875      throw new FacebookSignedRequestParsingException(format(
876        "Signed request '%s' is expected to be signature and payload strings separated by a '.'", signedRequest));
877    }
878
879    String encodedSignature = signedRequestTokens[0];
880    String urlDecodedSignature = urlDecodeSignedRequestToken(encodedSignature);
881    byte[] signature = decodeBase64(urlDecodedSignature);
882
883    String encodedPayload = signedRequestTokens[1];
884    String urlDecodedPayload = urlDecodeSignedRequestToken(encodedPayload);
885    String payload = StringUtils.toString(decodeBase64(urlDecodedPayload));
886
887    // Convert payload to a JsonObject so we can pull algorithm data out of it
888    JsonObject payloadObject = getJsonMapper().toJavaObject(payload, JsonObject.class);
889
890    if (!payloadObject.has("algorithm")) {
891      throw new FacebookSignedRequestParsingException("Unable to detect algorithm used for signed request");
892    }
893
894    String algorithm = payloadObject.getString("algorithm");
895
896    if (!verifySignedRequest(appSecret, algorithm, encodedPayload, signature)) {
897      throw new FacebookSignedRequestVerificationException(
898        "Signed request verification failed. Are you sure the request was made for the app identified by the app secret you provided?");
899    }
900
901    // Marshal to the user's preferred type.
902    // If the user asked for a JsonObject, send back the one we already parsed.
903    return objectType.equals(JsonObject.class) ? (T) payloadObject : getJsonMapper().toJavaObject(payload, objectType);
904  }

939  /**
940   * Verifies that the signed request is really from Facebook.
941   * 
942   * @param appSecret
943   *          The secret for the app that can verify this signed request.
944   * @param algorithm
945   *          Signature algorithm specified by FB in the decoded payload.
946   * @param encodedPayload
947   *          The encoded payload used to generate a signature for comparison against the provided {@code signature}.
948   * @param signature
949   *          The decoded signature extracted from the signed request. Compared against a signature generated from
950   *          {@code encodedPayload}.
951   * @return {@code true} if the signed request is verified, {@code false} if not.
952   */
953  protected boolean verifySignedRequest(String appSecret, String algorithm, String encodedPayload, byte[] signature) {
954    verifyParameterPresence("appSecret", appSecret);
955    verifyParameterPresence("algorithm", algorithm);
956    verifyParameterPresence("encodedPayload", encodedPayload);
957    verifyParameterPresence("signature", signature);
958
959    // Normalize algorithm name...FB calls it differently than Java does
960    if ("HMAC-SHA256".equalsIgnoreCase(algorithm)) {
961      algorithm = "HMACSHA256";
962    }
963
964    try {
965      Mac mac = Mac.getInstance(algorithm);
966      mac.init(new SecretKeySpec(toBytes(appSecret), algorithm));
967      byte[] payloadSignature = mac.doFinal(toBytes(encodedPayload));
968      return Arrays.equals(signature, payloadSignature);
969    } catch (Exception e) {
970      throw new FacebookSignedRequestVerificationException("Unable to perform signed request verification", e);
971    }
972  }

1024   * Coordinates the process of executing the API request GET/POST and processing the response we receive from the
1025   * endpoint.
1026   * 
1027   * @param endpoint
1028   *          Facebook Graph API endpoint.
1029   * @param executeAsPost
1030   *          {@code true} to execute the web request as a {@code POST}, {@code false} to execute as a {@code GET}.
1031   * @param executeAsDelete
1032   *          {@code true} to add a special 'treat this request as a {@code DELETE}' parameter.
1033   * @param binaryAttachments
1034   *          A list of binary files to include in a {@code POST} request. Pass {@code null} if no attachment should be
1035   *          sent.
1036   * @param parameters
1037   *          Arbitrary number of parameters to send along to Facebook as part of the API call.
1038   * @return The JSON returned by Facebook for the API call.
1039   * @throws FacebookException
1040   *           If an error occurs while making the Facebook API POST or processing the response.
1041   */
1042  protected String makeRequest(String endpoint, final boolean executeAsPost, final boolean executeAsDelete,
1043      final List<BinaryAttachment> binaryAttachments, Parameter... parameters) {
1044    verifyParameterLegality(parameters);
1045
1046    if (executeAsDelete && isHttpDeleteFallback()) {
1047      parameters = parametersWithAdditionalParameter(Parameter.with(METHOD_PARAM_NAME, "delete"), parameters);
1048    }
1049
1050    if (!endpoint.startsWith("/")) {
1051      endpoint = "/" + endpoint;
1052    }
1053
1054    final String fullEndpoint =
1055        createEndpointForApiCall(endpoint, binaryAttachments != null && !binaryAttachments.isEmpty());
1056    final String parameterString = toParameterString(parameters);
1057
1058    return makeRequestAndProcessResponse(new Requestor() {
1059      /**
1060       * @see com.restfb.DefaultFacebookClient.Requestor#makeRequest()
1061       */
1062      @Override
1063      public Response makeRequest() throws IOException {
1064        if (executeAsDelete && !isHttpDeleteFallback()) {
1065          return webRequestor.executeDelete(fullEndpoint + "?" + parameterString);
1066        } else {
1067          return executeAsPost
1068              ? webRequestor.executePost(fullEndpoint, parameterString,
1069                binaryAttachments == null ? null
1070                    : binaryAttachments.toArray(new BinaryAttachment[binaryAttachments.size()]))
1071              : webRequestor.executeGet(fullEndpoint + "?" + parameterString);
1072        }
1073      }
1074    });
1075  }

1112  protected String makeRequestAndProcessResponse(Requestor requestor) {
1113    Response response;
1114
1115    // Perform a GET or POST to the API endpoint
1116    try {
1117      response = requestor.makeRequest();
1118    } catch (Exception t) {
1119      throw new FacebookNetworkException("Facebook request failed", t);
1120    }
1121
1122    // If we get any HTTP response code other than a 200 OK or 400 Bad Request
1123    // or 401 Not Authorized or 403 Forbidden or 404 Not Found or 500 Internal
1124    // Server Error or 302 Not Modified
1125    // throw an exception.
1126    if (HTTP_OK != response.getStatusCode() && HTTP_BAD_REQUEST != response.getStatusCode()
1127        && HTTP_UNAUTHORIZED != response.getStatusCode() && HTTP_NOT_FOUND != response.getStatusCode()
1128        && HTTP_INTERNAL_ERROR != response.getStatusCode() && HTTP_FORBIDDEN != response.getStatusCode()
1129        && HTTP_NOT_MODIFIED != response.getStatusCode()) {
1130      throw new FacebookNetworkException("Facebook request failed", response.getStatusCode());
1131    }
1132
1133    String json = response.getBody();
1134
1135    // If the response contained an error code, throw an exception.
1136    throwFacebookResponseStatusExceptionIfNecessary(json, response.getStatusCode());
1137
1138    // If there was no response error information and this was a 500 or 401
1139    // error, something weird happened on Facebook's end. Bail.
1140    if (HTTP_INTERNAL_ERROR == response.getStatusCode() || HTTP_UNAUTHORIZED == response.getStatusCode()) {
1141      throw new FacebookNetworkException("Facebook request failed", response.getStatusCode());
1142    }
1143
1144    return json;
1145  }

1147  /**
1148   * Throws an exception if Facebook returned an error response. Using the Graph API, it's possible to see both the new
1149   * Graph API-style errors as well as Legacy API-style errors, so we have to handle both here. This method extracts
1150   * relevant information from the error JSON and throws an exception which encapsulates it for end-user consumption.
1151   * <p>
1152   * For Graph API errors:
1153   * <p>
1154   * If the {@code error} JSON field is present, we've got a response status error for this API call.
1155   * <p>
1156   * For Legacy errors (e.g. FQL):
1157   * <p>
1158   * If the {@code error_code} JSON field is present, we've got a response status error for this API call.
1159   * 
1160   * @param json
1161   *          The JSON returned by Facebook in response to an API call.
1162   * @param httpStatusCode
1163   *          The HTTP status code returned by the server, e.g. 500.
1164   * @throws FacebookGraphException
1165   *           If the JSON contains a Graph API error response.
1166   * @throws FacebookResponseStatusException
1167   *           If the JSON contains an Legacy API error response.
1168   * @throws FacebookJsonMappingException
1169   *           If an error occurs while processing the JSON.
1170   */
1171  protected void throwFacebookResponseStatusExceptionIfNecessary(String json, Integer httpStatusCode) {
1172    try {
1173      skipResponseStatusExceptionParsing(json);
1174
1175      // If we have a legacy exception, throw it.
1176      throwLegacyFacebookResponseStatusExceptionIfNecessary(json, httpStatusCode);
1177
1178      // If we have a batch API exception, throw it.
1179      throwBatchFacebookResponseStatusExceptionIfNecessary(json, httpStatusCode);
1180
1181      JsonObject errorObject = new JsonObject(json);
1182
1183      if (!errorObject.has(ERROR_ATTRIBUTE_NAME)) {
1184        return;
1185      }
1186
1187      JsonObject innerErrorObject = errorObject.getJsonObject(ERROR_ATTRIBUTE_NAME);
1188
1189      // If there's an Integer error code, pluck it out.
1190      Integer errorCode = innerErrorObject.has(ERROR_CODE_ATTRIBUTE_NAME)
1191          ? toInteger(innerErrorObject.getString(ERROR_CODE_ATTRIBUTE_NAME)) : null;
1192      Integer errorSubcode = innerErrorObject.has(ERROR_SUBCODE_ATTRIBUTE_NAME)
1193          ? toInteger(innerErrorObject.getString(ERROR_SUBCODE_ATTRIBUTE_NAME)) : null;
1194
1195      throw graphFacebookExceptionMapper.exceptionForTypeAndMessage(errorCode, errorSubcode, httpStatusCode,
1196        innerErrorObject.optString(ERROR_TYPE_ATTRIBUTE_NAME), innerErrorObject.getString(ERROR_MESSAGE_ATTRIBUTE_NAME),
1197        innerErrorObject.optString(ERROR_USER_TITLE_ATTRIBUTE_NAME),
1198        innerErrorObject.optString(ERROR_USER_MSG_ATTRIBUTE_NAME),
1199              innerErrorObject.optBoolean(ERROR_IS_TRANSIENT_NAME, false),
1200
1201              errorObject);
1202    } catch (JsonException e) {
1203      throw new FacebookJsonMappingException("Unable to process the Facebook API response", e);
1204    } catch (ResponseErrorJsonParsingException ex) {
1205      if (CLIENT_LOGGER.isTraceEnabled()) {
1206        CLIENT_LOGGER.trace("caught ResponseErrorJsonParsingException - ignoring", ex);
1207      }
1208    }
1209  }

1274    public FacebookException exceptionForTypeAndMessage(Integer errorCode, Integer errorSubcode, Integer httpStatusCode,

1279          errorUserMessage, isTransient, rawError);
1280      }
1281
1282      if ("QueryParseException".equals(type)) {
1283        return new FacebookQueryParseException(type, message, errorCode, errorSubcode, httpStatusCode, errorUserTitle,
1284          errorUserMessage, isTransient, rawError);
1285      }
1286
1287      // Don't recognize this exception type? Just go with the standard
1288      // FacebookGraphException.
1289      return new FacebookGraphException(type, message, errorCode, errorSubcode, httpStatusCode, errorUserTitle,
1290        errorUserMessage, isTransient, rawError);
1291    }
1292  }

1307  /**
1308   * Generate the parameter string to be included in the Facebook API request.
1309   * 
1310   * @param withJsonParameter
1311   *          add additional parameter format with type json
1312   * @param parameters
1313   *          Arbitrary number of extra parameters to include in the request.
1314   * @return The parameter string to include in the Facebook API request.
1315   * @throws FacebookJsonMappingException
1316   *           If an error occurs when building the parameter string.
1317   */
1318  protected String toParameterString(boolean withJsonParameter, Parameter... parameters) {
1319    if (!isBlank(accessToken)) {
1320      parameters = parametersWithAdditionalParameter(Parameter.with(ACCESS_TOKEN_PARAM_NAME, accessToken), parameters);
1321    }
1322
1323    if (!isBlank(accessToken) && !isBlank(appSecret)) {
1324      parameters = parametersWithAdditionalParameter(
1325        Parameter.with(APP_SECRET_PROOF_PARAM_NAME, obtainAppSecretProof(accessToken, appSecret)), parameters);
1326    }
1327
1328    if (withJsonParameter) {
1329      parameters = parametersWithAdditionalParameter(Parameter.with(FORMAT_PARAM_NAME, "json"), parameters);
1330    }
1331
1332    StringBuilder parameterStringBuilder = new StringBuilder();
1333    boolean first = true;
1334
1335    for (Parameter parameter : parameters) {
1336      if (first) {
1337        first = false;
1338      } else {
1339        parameterStringBuilder.append("&");
1340      }
1341
1342      parameterStringBuilder.append(urlEncode(parameter.name));
1343      parameterStringBuilder.append("=");
1344      parameterStringBuilder.append(urlEncodedValueForParameterName(parameter.name, parameter.value));
1345    }
1346
1347    return parameterStringBuilder.toString();
1348  }

1350  /**
1351   * @see com.restfb.BaseFacebookClient#createEndpointForApiCall(java.lang.String,boolean)
1352   */
1353  @Override
1354  protected String createEndpointForApiCall(String apiCall, boolean hasAttachment) {
1355    while (apiCall.startsWith("/"))
1356      apiCall = apiCall.substring(1);
1357
1358    String baseUrl = getFacebookGraphEndpointUrl();
1359
1360    if (readOnlyApiCalls.contains(apiCall)) {
1361      baseUrl = getFacebookReadOnlyEndpointUrl();
1362    } else if (hasAttachment && (apiCall.endsWith("/videos") || apiCall.endsWith("/advideos"))) {
1363      baseUrl = getFacebookGraphVideoEndpointUrl();
1364    } else if (apiCall.endsWith("logout.php")) {
1365      baseUrl = FACEBOOK_ENDPOINT_URL;
1366    }
1367
1368    return format("%s/%s", baseUrl, apiCall);
1369  }
