040/**
041 * Default implementation of a service that sends HTTP requests to the Facebook API endpoint.
042 * 
043 * @author <a href="http://restfb.com">Mark Allen</a>

045public class DefaultWebRequestor implements WebRequestor {
046  /**
047   * Arbitrary unique boundary marker for multipart {@code POST}s.
048   */
049  private static final String MULTIPART_BOUNDARY = "**boundarystringwhichwill**neverbeencounteredinthewild**";
050
051  /**
052   * Line separator for multipart {@code POST}s.
053   */
054  private static final String MULTIPART_CARRIAGE_RETURN_AND_NEWLINE = "\r\n";
055
056  /**
057   * Hyphens for multipart {@code POST}s.
058   */
059  private static final String MULTIPART_TWO_HYPHENS = "--";
060
061  /**
062   * Default buffer size for multipart {@code POST}s.
063   */
064  private static final int MULTIPART_DEFAULT_BUFFER_SIZE = 8192;
065
066  /**
067   * By default, how long should we wait for a response (in ms)?
068   */
069  private static final int DEFAULT_READ_TIMEOUT_IN_MS = 180000;

102  /**
103   * @throws java.io.IOException
104   * @see com.restfb.WebRequestor#executePost(java.lang.String, java.lang.String, com.restfb.BinaryAttachment[])
105   */
106  @Override
107  public Response executePost(String url, String parameters, BinaryAttachment... binaryAttachments) throws IOException {
108    if (binaryAttachments == null) {
109      binaryAttachments = new BinaryAttachment[0];
110    }
111
112    if (HTTP_LOGGER.isDebugEnabled()) {
113      HTTP_LOGGER.debug("Executing a POST to " + url + " with parameters "
114          + (binaryAttachments.length > 0 ? "" : "(sent in request body): ") + UrlUtils.urlDecode(parameters)
115          + (binaryAttachments.length > 0 ? " and " + binaryAttachments.length + " binary attachment[s]." : ""));
116    }
117
118    HttpURLConnection httpUrlConnection = null;
119    OutputStream outputStream = null;
120    InputStream inputStream = null;
121
122    try {
123      httpUrlConnection = openConnection(new URL(url + (binaryAttachments.length > 0 ? "?" + parameters : "")));
124      httpUrlConnection.setReadTimeout(DEFAULT_READ_TIMEOUT_IN_MS);
125
126      // Allow subclasses to customize the connection if they'd like to - set
127      // their own headers, timeouts, etc.
128      customizeConnection(httpUrlConnection);
129
130      httpUrlConnection.setRequestMethod("POST");
131      httpUrlConnection.setDoOutput(true);
132      httpUrlConnection.setUseCaches(false);
133
134      if (binaryAttachments.length > 0) {
135        httpUrlConnection.setRequestProperty("Connection", "Keep-Alive");
136        httpUrlConnection.setRequestProperty("Content-Type", "multipart/form-data;boundary=" + MULTIPART_BOUNDARY);
137      }
138
139      httpUrlConnection.connect();
140      outputStream = httpUrlConnection.getOutputStream();
141
142      // If we have binary attachments, the body is just the attachments and the
143      // other parameters are passed in via the URL.
144      // Otherwise the body is the URL parameter string.
145      if (binaryAttachments.length > 0) {
146        for (BinaryAttachment binaryAttachment : binaryAttachments) {
147          StringBuilder stringBuilder = new StringBuilder();
148
149          stringBuilder.append(MULTIPART_TWO_HYPHENS).append(MULTIPART_BOUNDARY)
150            .append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).append("Content-Disposition: form-data; name=\"")
151            .append(createFormFieldName(binaryAttachment)).append("\"; filename=\"")
152            .append(binaryAttachment.getFilename()).append("\"");
153
154          if (binaryAttachment.getContentType() != null && binaryAttachment.getContentType().length() > 0) {
155            stringBuilder.append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).append("Content-Type: ")
156              .append(binaryAttachment.getContentType());
157          }
158
159          stringBuilder.append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE);
160
161          outputStream.write(stringBuilder.toString().getBytes(StringUtils.ENCODING_CHARSET));
162
163          write(binaryAttachment.getData(), outputStream, MULTIPART_DEFAULT_BUFFER_SIZE);
164
165          outputStream.write((MULTIPART_CARRIAGE_RETURN_AND_NEWLINE + MULTIPART_TWO_HYPHENS + MULTIPART_BOUNDARY
166              + MULTIPART_TWO_HYPHENS + MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).getBytes(StringUtils.ENCODING_CHARSET));
167        }
168      } else {
169        outputStream.write(parameters.getBytes(StringUtils.ENCODING_CHARSET));
170      }
171
172      if (HTTP_LOGGER.isDebugEnabled()) {
173        HTTP_LOGGER.debug(format("Response headers: %s", httpUrlConnection.getHeaderFields()));
174      }
175
176      fillHeaderAndDebugInfo(httpUrlConnection);
177
178      try {
179        inputStream = httpUrlConnection.getResponseCode() != HttpURLConnection.HTTP_OK
180            ? httpUrlConnection.getErrorStream() : httpUrlConnection.getInputStream();
181      } catch (IOException e) {
182        HTTP_LOGGER.warn(format("An error occurred while POSTing to %s:", url), e);
183      }
184
185      return new Response(httpUrlConnection.getResponseCode(), StringUtils.fromInputStream(inputStream));
186    } finally {
187      if (autocloseBinaryAttachmentStream && binaryAttachments.length > 0) {
188        for (BinaryAttachment binaryAttachment : binaryAttachments) {
189          closeQuietly(binaryAttachment.getData());
190        }
191      }
192
193      closeQuietly(outputStream);
194      closeQuietly(httpUrlConnection);
195    }
196  }

198  /**
199   * Given a {@code url}, opens and returns a connection to it.

226  /**
227   * Attempts to cleanly close a resource, swallowing any exceptions that might occur since there's no way to recover
228   * anyway.
229   * <p>
230   * It's OK to pass {@code null} in, this method will no-op in that case.
231   * 
232   * @param closeable
233   *          The resource to close.
234   */
235  protected void closeQuietly(Closeable closeable) {
236    if (closeable == null) {
237      return;
238    }
239    try {
240      closeable.close();
241    } catch (Exception t) {
242      HTTP_LOGGER.warn(format("Unable to close %s: ", closeable), t);
243    }
244  }
245
246  /**
247   * Attempts to cleanly close an {@code HttpURLConnection}, swallowing any exceptions that might occur since there's no
248   * way to recover anyway.
249   * <p>
250   * It's OK to pass {@code null} in, this method will no-op in that case.
251   * 
252   * @param httpUrlConnection
253   *          The connection to close.
254   */
255  protected void closeQuietly(HttpURLConnection httpUrlConnection) {
256    if (httpUrlConnection == null) {
257      return;
258    }
259    try {
260      httpUrlConnection.disconnect();
261    } catch (Exception t) {
262      HTTP_LOGGER.warn(format("Unable to disconnect %s: ", httpUrlConnection), t);
263    }
264  }

266  /**
267   * Writes the contents of the {@code source} stream to the {@code destination} stream using the given
268   * {@code bufferSize}.
269   * 
270   * @param source
271   *          The source stream to copy from.
272   * @param destination
273   *          The destination stream to copy to.
274   * @param bufferSize
275   *          The size of the buffer to use during the copy operation.
276   * @throws IOException
277   *           If an error occurs when reading from {@code source} or writing to {@code destination}.
278   * @throws NullPointerException
279   *           If either {@code source} or @{code destination} is {@code null}.
280   */
281  protected void write(InputStream source, OutputStream destination, int bufferSize) throws IOException {
282    if (source == null || destination == null) {
283      throw new NullPointerException("Must provide non-null source and destination streams.");
284    }
285
286    int read;
287    byte[] chunk = new byte[bufferSize];
288    while ((read = source.read(chunk)) > 0)
289      destination.write(chunk, 0, read);
290  }

292  /**
293   * Creates the form field name for the binary attachment filename by stripping off the file extension - for example,
294   * the filename "test.png" would return "test".
295   * 
296   * @param binaryAttachment
297   *          The binary attachment for which to create the form field name.
298   * @return The form field name for the given binary attachment.
299   */
300  protected String createFormFieldName(BinaryAttachment binaryAttachment) {
301    if (binaryAttachment.getFieldName() != null) {
302      return binaryAttachment.getFieldName();
303    }
304
305    String name = binaryAttachment.getFilename();
306    int fileExtensionIndex = name.lastIndexOf('.');
307    return fileExtensionIndex > 0 ? name.substring(0, fileExtensionIndex) : name;
308  }

349  private Response execute(String url, HttpMethod httpMethod) throws IOException {
350    if (HTTP_LOGGER.isDebugEnabled()) {
351      HTTP_LOGGER.debug(format("Making a %s request to %s", httpMethod.name(), url));
352    }
353
354    HttpURLConnection httpUrlConnection = null;
355
356    try {
357      httpUrlConnection = openConnection(new URL(url));
358      httpUrlConnection.setReadTimeout(DEFAULT_READ_TIMEOUT_IN_MS);
359      httpUrlConnection.setUseCaches(false);
360      httpUrlConnection.setRequestMethod(httpMethod.name());
361
362      // Allow subclasses to customize the connection if they'd like to - set
363      // their own headers, timeouts, etc.
364      customizeConnection(httpUrlConnection);
365
366      httpUrlConnection.connect();
367
368      if (HTTP_LOGGER.isTraceEnabled()) {
369        HTTP_LOGGER.trace(format("Response headers: %s", httpUrlConnection.getHeaderFields()));
370      }
371
372      fillHeaderAndDebugInfo(httpUrlConnection);
373
374      Response response = fetchResponse(httpUrlConnection);
375
376      if (HTTP_LOGGER.isDebugEnabled()) {
377        HTTP_LOGGER.debug(format("Facebook responded with %s", response));
378      }
379
380      return response;
381    } finally {
382      closeQuietly(httpUrlConnection);
383    }
384  }
385
386  protected void fillHeaderAndDebugInfo(HttpURLConnection httpUrlConnection) {
387    currentHeaders = Collections.unmodifiableMap(httpUrlConnection.getHeaderFields());
388
389    String usedApiVersion = StringUtils.trimToEmpty(httpUrlConnection.getHeaderField("facebook-api-version"));
390    if (HTTP_LOGGER.isDebugEnabled()) {
391      HTTP_LOGGER.debug(format("Facebook used the API %s to answer your request", usedApiVersion));
392    }
393
394    String fbTraceId = StringUtils.trimToEmpty(httpUrlConnection.getHeaderField("x-fb-trace-id"));
395    String fbRev = StringUtils.trimToEmpty(httpUrlConnection.getHeaderField("x-fb-rev"));
396    String fbDebug = StringUtils.trimToEmpty(httpUrlConnection.getHeaderField("x-fb-debug"));
397    String fbAppUsage = StringUtils.trimToEmpty(httpUrlConnection.getHeaderField("x-app-usage"));
398    String fbPageUsage = StringUtils.trimToEmpty(httpUrlConnection.getHeaderField("x-page-usage"));
399    Version usedVersion = Version.getVersionFromString(usedApiVersion);
400    debugHeaderInfo = new DebugHeaderInfo(fbDebug, fbRev, fbTraceId, usedVersion, fbAppUsage, fbPageUsage);
401  }
402
403  protected Response fetchResponse(HttpURLConnection httpUrlConnection) throws IOException {
404    InputStream inputStream = null;
405    try {
406      inputStream = httpUrlConnection.getResponseCode() != HttpURLConnection.HTTP_OK
407          ? httpUrlConnection.getErrorStream() : httpUrlConnection.getInputStream();
408    } catch (IOException e) {
409      HTTP_LOGGER.warn(format("An error occurred while making a %s request to %s:",
410        httpUrlConnection.getRequestMethod(), httpUrlConnection.getURL()), e);
411    }
412
413    return new Response(httpUrlConnection.getResponseCode(), StringUtils.fromInputStream(inputStream));
414  }
