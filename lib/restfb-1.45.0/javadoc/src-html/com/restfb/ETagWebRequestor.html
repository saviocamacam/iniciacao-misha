022package com.restfb;
023
024import static java.net.HttpURLConnection.HTTP_NOT_MODIFIED;
025
026import com.restfb.util.SoftHashMap;
027
028import java.io.IOException;
029import java.net.HttpURLConnection;
030import java.util.Collections;
031import java.util.Map;
032
033/**
034 * WebRequestor with ETag-support.
035 *
036 * <p>
037 * The {@see ETagWebRequestor} caches all <tt>GET</tt>-requests with an ETag header field in a {@see SoftHashMap} and
038 * uses the ETag on the next request as <code>If-None-Match</code> header field if the same URL is requested.
039 * </p>
040 *
041 * <p>
042 * Is the response status code 304 (NOT MODIFIED) the old response from cache is used.
043 * </p>
044 *
045 * <p>
046 * <strong>Attention:</strong> even 304 responses count as request at Facebook and so they count against the throttling
047 * limits. Facebook suggests to use them for data that change only frequently
048 * </p>
049 *
050 * <p>
051 * Further information regarding ETag at Facebook can be found here:
052 * <a href="https://developers.facebook.com/blog/post/627/">https://developers.facebook.com/blog/post/627/</a>
053 * </p>
054 *
055 * <p>
056 * <strong>Attention 2</strong>: If excessively used with a lot of URLs, the {@see SoftHashMap} can lead to a
057 * performance degradation
058 * </p>
059 */
060public class ETagWebRequestor extends DefaultWebRequestor {
061
062  private final Map<String, ETagResponse> etagCache =
063      Collections.synchronizedMap(new SoftHashMap<String, ETagResponse>());
064  private final ThreadLocal<ETagResponse> currentETagRespThreadLocal = new ThreadLocal<ETagResponse>();
065  private volatile boolean useCache = true;
066
067  @Override
068  protected void customizeConnection(HttpURLConnection connection) {
069    if (isUseCache() && connection.getRequestMethod().equals(HttpMethod.GET.name())) {
070      ETagResponse resp = etagCache.get(connection.getURL().toString());
071      if (resp != null) {
072        currentETagRespThreadLocal.set(resp);
073        connection.addRequestProperty("If-None-Match", resp.getEtag());
074      }
075    }
076  }
077
078  @Override
079  protected Response fetchResponse(HttpURLConnection httpUrlConnection) throws IOException {
080    try {
081      if (httpUrlConnection.getRequestMethod().equals(HttpMethod.GET.name())) {
082        if (httpUrlConnection.getResponseCode() == HTTP_NOT_MODIFIED && currentETagRespThreadLocal.get() != null) {
083          ETagResponse etagResp = currentETagRespThreadLocal.get();
084          return new Response(httpUrlConnection.getResponseCode(), etagResp.getBody());
085        } else {
086          Response resp = super.fetchResponse(httpUrlConnection);
087          if (httpUrlConnection.getHeaderField("ETag") != null) {
088            etagCache.put(httpUrlConnection.getURL().toString(),
089              new ETagResponse(httpUrlConnection.getHeaderField("ETag"), resp.getBody()));
090          }
091          return resp;
092        }
093      } else {
094        return super.fetchResponse(httpUrlConnection);
095      }
096    } finally {
097      currentETagRespThreadLocal.remove();
098    }
099  }
100
101  /**
102   * return if cache is used.
103   * 
104   * @return <code>true</code> if ETag-Cache is used, <code>false</code> if not
105   */
106  public boolean isUseCache() {
107    return this.useCache;
108  }
109
110  /**
111   * activate/deactivate the ETag-Cache for the next request.
112   *
113   * <p>
114   * when deactivated, the ETag-Cache is *not* deleted
115   * </p>
116   *
117   * @param useCache
118   */
119  public void setUseCache(boolean useCache) {
120    this.useCache = useCache;
121  }
122
123  private class ETagResponse {
124
125    public ETagResponse(String etag, String body) {
126      this.etag = etag;
127      this.body = body;
128    }
129
130    private final String etag;
131    private final String body;
132
133    public String getEtag() {
134      return etag;
135    }
136
137    public String getBody() {
138      return body;
139    }
140  }
141
142}
