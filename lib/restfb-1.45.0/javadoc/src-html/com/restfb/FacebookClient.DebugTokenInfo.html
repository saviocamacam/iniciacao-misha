049/**
050 * Specifies how a <a href="http://developers.facebook.com/docs/api">Facebook Graph API</a> client must operate.
051 * <p>
052 * Projects that need to access the <a href="http://developers.facebook.com/docs/reference/rest/">old REST API</a>
053 * should use {@link LegacyFacebookClient} instead. You might choose to do this because you have a legacy codebase or
054 * you need functionality that is not yet available in the Graph API.
055 * <p>
056 * If you'd like to...
057 * 
058 * <ul>
059 * <li>Fetch an object: use {@link #fetchObject(String, Class, Parameter...)} or
060 * {@link #fetchObjects(List, Class, Parameter...)}</li>
061 * <li>Fetch a connection: use {@link #fetchConnection(String, Class, Parameter...)}</li>
062 * <li>Execute an FQL query: use {@link #executeFqlQuery(String, Class, Parameter...)} or
063 * {@link #executeFqlMultiquery(Map, Class, Parameter...)}</li>
064 * <li>Execute operations in batch: use {@link #executeBatch(BatchRequest...)} or {@link #executeBatch(List, List)}</li>
065 * <li>Publish data: use {@link #publish(String, Class, Parameter...)} or
066 * {@link #publish(String, Class, BinaryAttachment, Parameter...)}</li>
067 * <li>Delete an object: use {@link #deleteObject(String, Parameter...)}</li>
068 * </ul>
069 * 
070 * <p>
071 * You may also perform some common access token operations. If you'd like to...
072 * 
073 * <ul>
074 * <li>Extend the life of an access token: use {@link #obtainExtendedAccessToken(String, String, String)}</li>
075 * <li>Obtain an access token for use on behalf of an application instead of a user, use
076 * {@link #obtainAppAccessToken(String, String)}.</li>
077 * <li>Convert old-style session keys to OAuth access tokens: use
078 * {@link #convertSessionKeysToAccessTokens(String, String, String...)}</li>
079 * <li>Verify and extract data from a signed request: use {@link #parseSignedRequest(String, String, Class)}</li>
080 * </ul>
081 * 
082 * @author <a href="http://restfb.com">Mark Allen</a>
083 * @author Scott Hernandez
084 * @author Mattia Tommasone
085 * @author <a href="http://ex-nerd.com">Chris Petersen</a>
086 * @author Josef Gierbl
087 * @author Broc Seib
088 */
089public interface FacebookClient {

091   * Fetches a single <a href="http://developers.facebook.com/docs/reference/api/">Graph API object</a>, mapping the
092   * result to an instance of {@code objectType}.
093   *

095   *          Java type to map to.
096   * @param object
097   *          ID of the object to fetch, e.g. {@code "me"}.
098   * @param objectType
099   *          Object type token.
100   * @param parameters
101   *          URL parameters to include in the API call (optional).
102   * @return An instance of type {@code objectType} which contains the requested object's data.
103   * @throws FacebookException
104   *           If an error occurs while performing the API call.
105   */
106  <T> T fetchObject(String object, Class<T> objectType, Parameter... parameters);

537  /**
538   * Method to poll Facebook and fetch the Device Access Token.
539   *
540   * This method is used for the Graph API 2.5 or lower. If you need to use an newer version you should use
541   * {@link #obtainDeviceAccessToken(String)}
542   * 
543   * You have to use this method to check if the user confirms the authorization.
544   * 
545   * {@link FacebookOAuthException} can be thrown if the authorization is declined or still pending.
546   * 
547   * @param appId
548   *          The ID of your app, found in your app's dashboard.
549   * @param code
550   *          The device
551   * @return An extended access token for the given {@link AccessToken}.
552   * @throws com.restfb.exception.devicetoken.FacebookDeviceTokenCodeExpiredException
553   *           the {@link DeviceCode#getCode()} is expired, please fetch a new {@link DeviceCode}. Call
554   *           {@link #fetchDeviceCode(java.lang.String, com.restfb.scope.ScopeBuilder) }
555   * @throws com.restfb.exception.devicetoken.FacebookDeviceTokenPendingException
556   *           the user has not finished the authorisation process, yet. Please poll again later.
557   * @throws com.restfb.exception.devicetoken.FacebookDeviceTokenDeclinedException
558   *           the user declined the authorisation. You have to handle this problem.
559   * @throws com.restfb.exception.devicetoken.FacebookDeviceTokenSlowdownException
560   *           you tried too often to fetch the device access token. You have to use a larger interval

698  /**
699   * Represents an access token/expiration date pair.
700   * <p>
701   * Facebook returns these types when performing access token-related operations - see
702   * {@link com.restfb.FacebookClient#convertSessionKeysToAccessTokens(String, String, String...)},
703   * {@link com.restfb.FacebookClient#obtainAppAccessToken(String, String)}, and
704   * {@link com.restfb.FacebookClient#obtainExtendedAccessToken(String, String, String)} for details.
705   * 
706   * @author <a href="http://restfb.com">Mark Allen</a>
707   */
708  class AccessToken {
709    @Facebook("access_token")
710    private String accessToken;
711
712    @Facebook("expires_in")
713    private Long rawExpires;
714
715    private Long expires;
716
717    @Facebook("token_type")
718    private String tokenType;
719
720    /**
721     * Given a query string of the form {@code access_token=XXX} or {@code access_token=XXX&expires=YYY}, return an
722     * {@code AccessToken} instance.
723     * <p>
724     * The {@code queryString} is required to contain an {@code access_token} parameter with a non-{@code null} value.
725     * The {@code expires} value is optional and should be the number of seconds since the epoch. If the {@code expires}
726     * value cannot be parsed, the returned {@code AccessToken} will have a {@code null} {@code expires} value.
727     * 
728     * @param queryString
729     *          The Facebook query string out of which to parse an {@code AccessToken} instance.
730     * @return An {@code AccessToken} instance which corresponds to the given {@code queryString}.
731     * @throws IllegalArgumentException
732     *           If no {@code access_token} parameter is present in the query string.
733     * @since 1.6.10
734     */
735    public static AccessToken fromQueryString(String queryString) {
736      // Query string can be of the form 'access_token=XXX' or
737      // 'access_token=XXX&expires=YYY'
738      Map<String, List<String>> urlParameters = extractParametersFromQueryString(queryString);
739
740      String extendedAccessToken = null;
741      String tokenType = null;
742
743      if (urlParameters.containsKey("access_token")) {
744        extendedAccessToken = urlParameters.get("access_token").get(0);
745      }
746
747      if (urlParameters.containsKey("token_type")) {
748        tokenType = urlParameters.get("token_type").get(0);
749      }
750
751      if (extendedAccessToken == null) {
752        throw new IllegalArgumentException(format(
753          "Was expecting a query string of the form 'access_token=XXX' or 'access_token=XXX&expires=YYY'. Instead, the query string was '%s'",
754          queryString));
755      }
756
757      Long expires = null;
758
759      // If an expires or expires_in value was provided and it's a valid long, great - use it.
760      // Otherwise ignore it.
761      String rawExpires = null;
762
763      if (urlParameters.containsKey("expires")) {
764        rawExpires = urlParameters.get("expires").get(0);
765      }
766
767      if (urlParameters.containsKey("expires_in")) {
768        rawExpires = urlParameters.get("expires_in").get(0);
769      }
770
771      if (rawExpires != null) {
772        try {
773          expires = Long.valueOf(rawExpires);
774        } catch (NumberFormatException e) {}
775        if (expires != null) {
776          expires = new Date().getTime() + 1000L * expires;
777        }
778      }
779
780      AccessToken accessToken = new AccessToken();
781      accessToken.accessToken = extendedAccessToken;
782      accessToken.expires = expires;
783      accessToken.tokenType = tokenType;
784      return accessToken;
785    }
786
787    @Override
788    public int hashCode() {
789      return ReflectionUtils.hashCode(this);
790    }
791
792    @Override
793    public boolean equals(Object that) {
794      return ReflectionUtils.equals(this, that);
795    }
796
797    @Override
798    public String toString() {
799      return ReflectionUtils.toString(this);
800    }
801
802    /**
803     * The access token's value.
804     * 
805     * @return The access token's value.
806     */
807    public String getAccessToken() {
808      return accessToken;
809    }
810
811    /**
812     * The date on which the access token expires.
813     * 
814     * @return The date on which the access token expires.
815     */
816    public Date getExpires() {
817      return expires == null ? null : new Date(expires);
818    }
819
820    /**
821     * The token type of this access token provided by Facebook
822     * 
823     * @return the access token type
824     */
825    public String getTokenType() {
826      return tokenType;
827    }
828
829    @JsonMapper.JsonMappingCompleted
830    void convertExpires() {
831      if (rawExpires != null) {
832        expires = new Date().getTime() + 1000L * rawExpires;
833      }
834    }
