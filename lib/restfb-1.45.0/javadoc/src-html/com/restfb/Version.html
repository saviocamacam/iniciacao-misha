022package com.restfb;
023
024public enum Version {
025
026  /**
027   * unversiond api
028   */
029  UNVERSIONED(null, false),
030
031  /**
032   * <tt>Graph API 1.0</tt>, available until April 30, 2015
033   * 
034   * @since April 21, 2010
035   * @deprecated use {@link Version#VERSION_2_3} instead
036   */
037  @Deprecated
038  VERSION_1_0("v1.0", false),
039
040  /**
041   * <tt>Graph API 2.0</tt>, available until August 7, 2016
042   *
043   * @since April 30, 2014
044   * @deprecated use {@link Version#VERSION_2_3} instead
045   */
046  @Deprecated
047  VERSION_2_0("v2.0", false),
048
049  /**
050   * <tt>Graph API 2.1</tt>, available until October 30, 2016
051   *
052   * @since August 7, 2014
053   * @deprecated use {@link Version#VERSION_2_3} instead
054   */
055  @Deprecated
056  VERSION_2_1("v2.1", false),
057
058  /**
059   * <tt>Graph API 2.2</tt>, available until March 25, 2017
060   *
061   * @since October 30, 2014
062   * @deprecated use {@link Version#VERSION_2_3} instead
063   */
064  @Deprecated
065  VERSION_2_2("v2.2", false),
066
067  /**
068   * <tt>Graph API 2.3</tt>, available until July 8, 2017
069   *
070   * @since March 25, 2015
071   */
072  VERSION_2_3("v2.3", false),
073
074  /**
075   * <tt>Graph API 2.4</tt>, available until October 7, 2017
076   *
077   * @since July 8, 2015
078   */
079  VERSION_2_4("v2.4", false),
080
081  /**
082   * <tt>Graph API 2.5</tt>, available until April 12, 2018
083   *
084   * @since October 7, 2015
085   */
086  VERSION_2_5("v2.5", false),
087
088  /**
089   * <tt>Graph API 2.6</tt>, available until July 13, 2018
090   *
091   * @since April 12, 2016
092   */
093  VERSION_2_6("v2.6", true),
094
095  /**
096   * <tt>Graph API 2.7</tt>, available until October 5, 2018
097   *
098   * @since July 13, 2016
099   */
100  VERSION_2_7("v2.7", true),
101
102  /**
103   * <tt>Graph API 2.8</tt>, available until April 18th, 2019
104   *
105   * @since October 5th, 2016
106   */
107  VERSION_2_8("v2.8", true),
108
109  /**
110   * <tt>Graph API 2.9</tt>, available until July 18th, 2019
111   *
112   * @since April 18th, 2017
113   */
114  VERSION_2_9("v2.9", true),
115
116  /**
117   * <tt>Graph API 2.10</tt>, available at least until July, 2019
118   *
119   * @since July 18th, 2017
120   */
121  VERSION_2_10("v2.10", true),
122
123  /**
124   * convenience enum to provide simple access to the latest supported Graph API Version.
125   * <p>
126   * the current version is <tt>Graph API 2.10</tt>
127   * </p>
128   */
129  LATEST("v2.10", true);
130
131  private final String urlElement;
132
133  private final boolean newDeviceTokenMethod;
134
135  Version(String urlElement, boolean newDeviceTokenMethod) {
136    this.urlElement = urlElement;
137    this.newDeviceTokenMethod = newDeviceTokenMethod;
138  }
139
140  public String getUrlElement() {
141    return this.urlElement;
142  }
143
144  public boolean isNewDeviceTokenMethod() {
145    return this.newDeviceTokenMethod;
146  }
147
148  public boolean isUrlElementRequired() {
149    return null != this.urlElement;
150  }
151
152  /**
153   * converts a String (for example the url paramter) into a Version object
154   * 
155   * @param urlElement
156   *          String that should
157   * @return the generated version
158   */
159  public static Version getVersionFromString(String urlElement) {
160    if (urlElement != null) {
161      for (Version v : Version.values()) {
162        if (urlElement.equals(v.getUrlElement())) {
163          return v;
164        }
165      }
166    }
167    return UNVERSIONED;
168  }
169}
