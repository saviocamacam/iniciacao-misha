022package com.restfb.logging;
023
024import com.restfb.exception.FacebookLoggerException;
025
026import java.lang.reflect.Constructor;
027
028/**
029 *
030 */
031public abstract class RestFBLogger {
032
033  private static Class LOGGER_CLASS = null;
034
035  public static final RestFBLogger HTTP_LOGGER;
036
037  public static final RestFBLogger MAPPER_LOGGER;
038
039  public static final RestFBLogger UTILS_LOGGER;
040
041  public static final RestFBLogger CLIENT_LOGGER;
042
043  public static final RestFBLogger VALUE_FACTORY_LOGGER;
044
045  static {
046    // define our logger class
047    Class loggerClass = com.restfb.logging.JulLogger.class;
048    LOGGER_CLASS = loggerClass;
049
050    // create the loggers
051    HTTP_LOGGER = getLoggerInstance("com.restfb.HTTP");
052    MAPPER_LOGGER = getLoggerInstance("com.restfb.DefaultJsonMapper");
053    UTILS_LOGGER = getLoggerInstance("com.restfb.UTILITY");
054    CLIENT_LOGGER = getLoggerInstance("com.restfb.DefaultFacebookClient");
055    VALUE_FACTORY_LOGGER = getLoggerInstance("com.restfb.types.webhook.ChangeValueFactory");
056  }
057
058  public static RestFBLogger getLoggerInstance(String logCategory) {
059    Object obj;
060    Class[] ctrTypes = new Class[] { String.class };
061    Object[] ctrArgs = new Object[] { logCategory };
062    try {
063      Constructor ctor = LOGGER_CLASS.getConstructor(ctrTypes);
064      obj = ctor.newInstance(ctrArgs);
065    } catch (Exception e) {
066      throw new FacebookLoggerException("logger cannot be created");
067    }
068
069    return (RestFBLogger) obj;
070  }
071
072  public abstract void trace(Object msg);
073
074  public abstract void trace(Object msg, Throwable thr);
075
076  public abstract void debug(Object msg);
077
078  public abstract void debug(Object msg, Throwable thr);
079
080  public abstract void info(Object msg);
081
082  public abstract void info(Object msg, Throwable thr);
083
084  public abstract void warn(Object msg);
085
086  public abstract void warn(Object msg, Throwable thr);
087
088  public abstract void error(Object msg);
089
090  public abstract void error(Object msg, Throwable thr);
091
092  public abstract void fatal(Object msg);
093
094  public abstract void fatal(Object msg, Throwable thr);
095
096  public abstract boolean isDebugEnabled();
097
098  public abstract boolean isInfoEnabled();
099
100  public abstract boolean isTraceEnabled();
101}
