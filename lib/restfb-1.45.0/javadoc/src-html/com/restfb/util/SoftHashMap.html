026package com.restfb.util;
027
028import java.io.Serializable;
029import java.lang.ref.Reference;
030import java.lang.ref.ReferenceQueue;
031import java.lang.ref.SoftReference;
032import java.util.*;
033
034public class SoftHashMap<K, V> extends AbstractMap<K, V>implements Serializable {
035  /** The internal HashMap that will hold the SoftReference. */
036  private final Map<K, SoftReference<V>> hash = new HashMap<K, SoftReference<V>>();
037
038  private final Map<SoftReference<V>, K> reverseLookup = new HashMap<SoftReference<V>, K>();
039
040  /** Reference queue for cleared SoftReference objects. */
041  private final ReferenceQueue<V> queue = new ReferenceQueue<V>();
042
043  @Override
044  public V get(Object key) {
045    expungeStaleEntries();
046    V result = null;
047    // We get the SoftReference represented by that key
048    SoftReference<V> softRef = hash.get(key);
049    if (softRef != null) {
050      // From the SoftReference we get the value, which can be
051      // null if it has been garbage collected
052      result = softRef.get();
053      if (result == null) {
054        // If the value has been garbage collected, remove the
055        // entry from the HashMap.
056        hash.remove(key);
057        reverseLookup.remove(softRef);
058      }
059    }
060    return result;
061  }
062
063  private void expungeStaleEntries() {
064    Reference<? extends V> sv;
065    while ((sv = queue.poll()) != null) {
066      hash.remove(reverseLookup.remove(sv));
067    }
068  }
069
070  @Override
071  public V put(K key, V value) {
072    expungeStaleEntries();
073    SoftReference<V> soft_ref = new SoftReference<V>(value, queue);
074    reverseLookup.put(soft_ref, key);
075    SoftReference<V> result = hash.put(key, soft_ref);
076    if (result == null) {
077      return null;
078    }
079    reverseLookup.remove(result);
080    return result.get();
081  }
082
083  @Override
084  public V remove(Object key) {
085    expungeStaleEntries();
086    SoftReference<V> result = hash.remove(key);
087    if (result == null) {
088      return null;
089    }
090    return result.get();
091  }
092
093  @Override
094  public void clear() {
095    hash.clear();
096    reverseLookup.clear();
097  }
098
099  @Override
100  public int size() {
101    expungeStaleEntries();
102    return hash.size();
103  }
104
105  /**
106   * Returns a copy of the key/values in the map at the point of calling. However, setValue still sets the value in the
107   * actual SoftHashMap.
108   */
109  @Override
110  public Set<Entry<K, V>> entrySet() {
111    expungeStaleEntries();
112    Set<Entry<K, V>> result = new LinkedHashSet<Entry<K, V>>();
113    for (final Entry<K, SoftReference<V>> entry : hash.entrySet()) {
114      final V value = entry.getValue().get();
115      if (value != null) {
116        result.add(new Entry<K, V>() {
117          @Override
118          public K getKey() {
119            return entry.getKey();
120          }
121
122          @Override
123          public V getValue() {
124            return value;
125          }
126
127          @Override
128          public V setValue(V v) {
129            entry.setValue(new SoftReference<V>(v, queue));
130            return value;
131          }
132        });
133      }
134    }
135    return result;
136  }
137}
